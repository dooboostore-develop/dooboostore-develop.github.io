Quick Start

Storage Operations

Handle localStorage and sessionStorage with ease:

localStorage와 sessionStorage를 쉽게 처리하세요:

import { StorageUtils } from '''@dooboostore/core-web/storage/StorageUtils'''; // Save an object to localStorage const user = { id: 1, name: '''John Doe''' }; StorageUtils.setLocalStorageItem('''currentUser''', user, window); // Retrieve and parse the object const savedUser = StorageUtils.getLocalStorageJsonItem<{ id: number, name: string }>('''currentUser''', window); console.log(savedUser?.name); // "John Doe" // Remove the item StorageUtils.removeLocalStorageItem('''currentUser''', window); // Session storage operations StorageUtils.setSessionStorageItem('''tempData''', { value: '''temporary''' }, window); const tempData = StorageUtils.getSessionStorageJsonItem('''tempData''', window);

Clipboard Operations

Copy and paste text with clipboard API:

클립보드 API로 텍스트를 복사하고 붙여넣기하세요:

import { ClipBoardUtils } from '''@dooboostore/core-web/clipboard/ClipBoardUtils'''; async function copyToClipboard(text: string) { try { await ClipBoardUtils.writeText(text, window); console.log('''Text copied successfully!'''); } catch (err) { console.error('''Failed to copy text: ''', err); } } async function readFromClipboard() { try { const text = await ClipBoardUtils.readText(window); console.log('''Clipboard content:''', text); } catch (err) { console.error('''Failed to read clipboard: ''', err); } } copyToClipboard('''Hello, world!''');

Element Loading

Load images and audio elements asynchronously:

이미지와 오디오 요소를 비동기적으로 로드하세요:

import { ElementUtils } from '''@dooboostore/core-web/element/ElementUtils'''; async function displayImage(src: string) { try { const imgElement = await ElementUtils.loadImage(src); document.body.appendChild(imgElement); console.log(`Image loaded: ${imgElement.width}x${imgElement.height}`); } catch (error) { console.error('''Failed to load image''', error); } } async function loadAudio(src: string) { try { const audioElement = await ElementUtils.loadAudio(src); audioElement.play(); } catch (error) { console.error('''Failed to load audio''', error); } } displayImage('''https://via.placeholder.com/150''');

Advanced Features

Image Conversion & Processing

Convert between different image formats and compress images:

다양한 이미지 형식 간 변환 및 이미지 압축:

import { ConvertUtils } from '''@dooboostore/core-web/convert/ConvertUtils'''; async function processImage(file: File) { try { // Convert File to ImageBitmap const imageBitmap = await ConvertUtils.toImageBitmap(file); // Compress image const compressedBlob = await ConvertUtils.compressImage(imageBitmap, { maxWidth: 800, type: '''image/jpeg''', quality: 0.8 }); // Convert to Base64 const base64 = await ConvertUtils.toBase64(compressedBlob, { type: '''image/jpeg''', quality: 0.8 }); // Create object URL for display const objectUrl = ConvertUtils.toObjectUrl(compressedBlob); console.log('''Processed image:''', { base64, objectUrl }); } catch (error) { console.error('''Image processing failed:''', error); } }

Dynamic Script Loading

Load external scripts and stylesheets dynamically:

외부 스크립트와 스타일시트를 동적으로 로드하세요:

import { ScriptUtils } from '''@dooboostore/core-web/script/ScriptUtils'''; async function loadExternalResources() { try { // Load external JavaScript await ScriptUtils.loadScript(document, '''https://cdn.jsdelivr.net/npm/lodash@4.17.21/lodash.min.js'''); // Load external CSS await ScriptUtils.loadStyleSheet(document, '''https://cdn.jsdelivr.net/npm/bootstrap@5.1.3/dist/css/bootstrap.min.css'''); // Load inline script await ScriptUtils.loadScriptBody(document, ` console.log('''Inline script loaded!'''); window.myCustomFunction = function() { return '''Hello from dynamic script!'''; }; `); console.log('''All resources loaded successfully!'''); } catch (error) { console.error('''Failed to load resources:''', error); } }

Environment Detection

Detect browser environment and mobile webviews:

브라우저 환경과 모바일 웹뷰를 감지하세요:

import { ValidUtils } from '''@dooboostore/core-web/valid/ValidUtils'''; function checkEnvironment() { // Check if running in browser if (ValidUtils.isBrowser()) { console.log('''Running in browser environment'''); // Check for mobile if (ValidUtils.isMobile()) { console.log('''Mobile device detected'''); } // Check for social network in-app browsers if (ValidUtils.isSocialNetworkServiceInappBrowser()) { console.log('''Social network in-app browser detected'''); } // Check for iOS webview if (ValidUtils.isWkWebview()) { console.log('''iOS WKWebView detected'''); } // Check for Android webview if (ValidUtils.isAndroidWebview()) { console.log('''Android WebView detected'''); } // Check if has parent window (iframe) if (ValidUtils.hasParentWindow(window)) { console.log('''Running inside iframe'''); } } else { console.log('''Not running in browser environment'''); } } checkEnvironment();

Advanced DOM querying and manipulation:

고급 DOM 쿼리 및 조작:

import { ElementUtils } from '''@dooboostore/core-web/element/ElementUtils'''; function manipulateDOM() { // Query elements const elements = ElementUtils.selectorElements('''.my-class'''); console.log(`Found ${elements.length} elements`); // Get element attributes as object const firstElement = elements[0]; if (firstElement) { const attributes = ElementUtils.getAttributeToObject(firstElement); console.log('''Element attributes:''', attributes); // Get element styles as object if (firstElement instanceof HTMLElement) { const styles = ElementUtils.getStyleToObject(firstElement); console.log('''Element styles:''', styles); } // Set multiple attributes ElementUtils.setAttributeAttr(firstElement, [ { name: '''data-processed''', value: '''true''' }, { name: '''aria-label''', value: '''Processed element''' } ]); // Remove attributes ElementUtils.removeAttribute(firstElement, ['''data-temp''', '''data-old''']); } }
