HTML Processing & Transformation

ÌîÑÎ°úÍ∑∏ÎûòÎ∞ç Î∞©ÏãùÏúºÎ°ú HTML ÏΩòÌÖêÏ∏† Ï≤òÎ¶¨ Î∞è Î≥ÄÌôò

import { DomParser } from
'@dooboostore/dom-parser';

function processHtml(html: string) {
    const parser = new DomParser(html);
    const document = parser.document;

    // Modify the title
    const title = document.querySelector('title');
    if (title) {
        title.textContent = 'Processed - ' + title.textContent;
    }

    // Add meta tags
    const head = document.querySelector('head');
    const metaDescription = document.createElement('meta');
    metaDescription.setAttribute('name', 'description');
    metaDescription.setAttribute('content', 'Processed by dom-parser');
    head?.appendChild(metaDescription);

    // Add CSS class to all paragraphs
    const paragraphs = document.querySelectorAll('p');
    paragraphs.forEach(p => {
        p.classList.add('processed');
    });

    // Wrap content in a container
    const body = document.body;
    const wrapper = document.createElement('div');
    wrapper.className = 'content-wrapper';

    // Move all body children to wrapper
    while (body.firstChild) {
        wrapper.appendChild(body.firstChild);
    }
    body.appendChild(wrapper);

    return document.documentElement.outerHTML;
}

        image,
        rating

});

return products;
}
